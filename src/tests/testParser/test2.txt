PRUEBAINTEGRAL {
    ## 1. DECLARACION DE VARIABLES (Tema 11 - Var/Inferencia) ##
    var A, B, RESULTADO;
    var CONTADOR;
    var FLOTANTE;

    ## 2. ASIGNACION SIMPLE Y ARITMETICA ENTERA (Tema 1) ##
    ## Nota: Usamos := para asignacion simple y sufijo I ##
    A := 10I;
    B := 5I;
    
    print("--- INICIO DE PRUEBAS ---");
    print("A vale (10):");
    print(A);

    ## 3. ESTRUCTURA IF-ELSE ##
    if (A > B) {
        print("Correcto: A es mayor que B");
    } else {
        print("Error: A no deberia ser menor");
    } endif;

    ## 4. CONVERSION EXPLICITA TOF (Tema 29) ##
    ## Probamos sumar un entero convertido con un flotante ##
    FLOTANTE := tof(A) + 2.5;
    print("Prueba TOF (12.5):");
    print(FLOTANTE);

    ## 5. CICLO FOR ASCENDENTE (Tema 15) ##
    ## El compilador debe detectar que 1 <= 3 y generar incremento ##
    print("For Ascendente (1 to 3):");
    for (CONTADOR from 1I to 3I) {
        print(CONTADOR);
    };

    ## 6. CICLO FOR DESCENDENTE (Tema 15) ##
    ## El compilador debe detectar que 3 > 1 y generar decremento ##
    print("For Descendente (3 to 1):");
    for (CONTADOR from 3I to 1I) {
        print(CONTADOR);
    };

    ## 7. ASIGNACION MULTIPLE (Tema 16) ##
    ## Nota: Usamos = para asignacion multiple segun tu gramatica ##
    print("Asignacion Multiple A,B = 1,2:");
    A, B = 1I, 2I;
    print(A);
    print(B);

    ## 8. PRUEBA DE LAMBDA (Tema 27) ##
    ## Definicion y ejecucion en linea ##
    print("Prueba Lambda (Imprime 100):");
    (int A) {if (A > 5I) print("A es mayor");}(10I) 
    print("--- FIN DE PRUEBAS ---");
}
